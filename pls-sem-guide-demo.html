Specifying the measurement models (or outer models)- SEMinR uses the constructs() function to specify the list of all construct measurement models. Within this list, we can then define various constructs:

# Create measurement model. compiles the list of constructs and their respective
#measurement model definitions. We must supply it with any number of individual
#composite(), interaction_term(), or higher_composite()
#constructs using their respective functions. Note that neither a dataset nor a structural
#model is specified in the measurement model stage, so we can reuse the measurement
#model object across different datasets and structural models.

simple_mm <- constructs(
composite("COMP", multi_items("comp_", 1:3)),
composite("LIKE", multi_items("like_", 1:3)),
composite("CUSA", single_item("cusa")),
composite("CUSL", multi_items("cusl_", 1:3)))

  mutate(watershed = "dh",
         date = mdy(date))

# Combine the 'fool_input' files (for 'fc' watershed)
fool_input_combined <- bind_rows(fool_input_2021, fool_input_2022) %>%
  mutate(watershed = "fc",
         date = ymd(date))
# For inputs_dh_combined
dh_input_combined <- dh_input_combined %>%
  arrange(date) %>%
  mutate(
    snowpack_3day_avg = rollapply(SWE.mm, 3, mean, fill = NA, align = "right") %>%
                        round(digits = 2),  # Adjust digits as needed
    melt_snow_interact = Input.mm * SWE.mm
  )

# For fool_input_combined
fool_input_combined <- fool_input_combined %>%
  arrange(date) %>%
  mutate(
    snowpack_3day_avg = rollapply(SWE.mm, 3, mean, fill = NA, align = "right")%>%
                        round(digits =2),
    melt_snow_interact = runoff_input * SWE.mm
  )
sample_fc <- sample_vege_fluro_topo %>%
  filter(watershed == "fc") %>%
  left_join(
    fool_input_combined %>%
      rename(Input.mm = runoff_input) %>%  # Rename before selecting and joining
      select(date, watershed, Input.mm, snowpack_3day_avg, melt_snow_interact),
    by = c("date", "watershed")
  )


# Join for the 'dh' watersheds (including 'lexen')
